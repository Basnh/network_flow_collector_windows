Run this PowerShell snippet on the target host to reconnect or update this agent configuration.

# Connect Agent to Central Hub
$AgentID = "{{ agent.agent_id }}"
$HubURL = "{{ request.url_root }}"
$Secret = "********************"

Invoke-WebRequest -Uri "$HubURL/api/agent/heartbeat" `
    -Method Post `
    -Headers @{"X-Agent-ID"=$AgentID; "X-Secret"=$Secret} `
    -Body (ConvertTo-Json @{ status="active"; hostname="$env:COMPUTERNAME" })
